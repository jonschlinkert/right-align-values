## Usage

**object**

```js
align({
  a: 'b', 
  c: 'dddddd', 
  e: 'fff', 
  g: 'hhhhh'
});
```

Results in:

```js
{
  a: '     b',
  c: 'dddddd',
  e: '   fff',
  g: ' hhhhh',
}
```

**array of objects**


```js
var align = require('{%= name %}');

var arr = [{a: 'b'}, {a: 'bb'}, {a: 'bbbb'}, {a: 'bbb'}, {a: 'bb'}];
align(arr, 'a');
```

Results in:

```js
[
  {a: '   b'},
  {a: '  bb'},
  {a: 'bbbb'},
  {a: ' bbb'},
  {a: '  bb'}
];
```
